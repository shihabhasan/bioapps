def color(name)
  return "" unless color?
  return "" unless ansi = ANSI[name.to_sym]
  "\e[#{ansi}m"
end

def color?
  return true if @@color_force
  return false if Foreman.windows?
  return false unless self.respond_to?(:isatty)
  self.isatty && ENV["TERM"]
end
